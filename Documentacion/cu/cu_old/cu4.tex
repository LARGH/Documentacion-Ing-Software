	\begin{UseCase}{CU4}{Actualizar datos personales.}{
		Resumen: Un representante podrá modificar sus datos personales a excepción del nombre completo.}
		\UCitem{Versión}{1.0}
		\UCitem{Autor}{Alejandro Edgar Mundo López.}
		\UCitem{Reviso}{Edgar Rodrigo Arredondo Basurto .}
		\UCitem{Estatus}{En corrección.}
		\UCitem{Fecha de último Estatus}{20 de Marzo del 2017.}
		\UCitem{Actor}{Representante}
		\UCitem{Propósito}{Editar los datos personales del representante, a excepción del nombre completo.}
		\UCitem{Entradas}{\begin{itemize}
		\item Teléfono: se escribe mediante el teclado.
		\item Correo: se escribe mediante el teclado.
		\item Foto: se selecciona la ruta de la misma.
		\item Contraseña: se escribe mediante el teclado.
		\item Nueva Contraseña: se escribe mediante el teclado.
		\end{itemize}. 
		}
		\UCitem{Salidas}{MSG4.1 Confirmación de la actalización}
		\UCitem{Postcondiciones}{Interna: Se actualizan los datos del representante.}
		\UCitem{Tipo}{Caso de uso primario}
		
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
	\UCpaso[\UCactor] Presiona el botón \IUbutton{Solicitar cambio de horario.} en la \IUref{IU1.0}{Pantalla de inicio del capitán}.
	\UCpaso El sistema válida que que el equipo este autorizado por el coordinador\Trayref{A}.
	\UCpaso Se muestra la \IUref{IU3.0}{Pantalla emergente de solicitud} para escribir el motivo de está.
	\UCpaso[\UCactor] Oprime el botón \IUbutton{Aceptar}.
	\UCpaso Se muestra una pantalla emergente el mensaje.{\bf MSG3}``Solicitud de camibio de hora enviada.''.
\Trayref{B}.
    \UCpaso[\UCactor] Oprime el botón \IUbutton{Aceptar}.

\end{UCtrayectoria}
