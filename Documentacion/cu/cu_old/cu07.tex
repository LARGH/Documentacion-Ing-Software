	\begin{UseCase}{CU7}{Consultar grupos y estadísticas}{
		Resumen: Un Usuario ingresa al sistema para consultar la información los grupos, así como las estadísticas del progreso de un torneo.	}
		\UCitem{Versión}{1.0}
		\UCitem{Actor}{Jugador de un equipo.}
		\UCitem{Propósito}{Conocer el progreso de un torneo o de un equipo en un torneo, así como los próximos partidos.}
		\UCitem{Entradas}{Filtros: tipo de torneo, estadísticas, grupos o próximos partidos.}
		\UCitem{Origen}{IU1.0 Pantalla de inicio.}
		\UCitem{Salidas}{Estadísticas, grupos o próximos partidos de un torneo.}
		\UCitem{Precondiciones}{Existe un torneo en progreso.}
		\UCitem{Tipo}{Caso de uso primario.}
		\UCitem{Observaciones}{Cualquier usuario que ingrese al sistema puede realizar consultas de estadísticas o grupos y próximos partidos. Agregar pantalla IU0X.}

		\UCitem{Reviso}{Landa Aguirre Rafael.}
