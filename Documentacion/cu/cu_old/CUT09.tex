	\begin{UseCase}{CUT9}{Actualizar fecha de partido}{
		Resumen: Este caso de uso permite al Coordinador, realizar un cambio en la fecha de cualquier partido que se vaya a jugar en un tiempo de aproximadamente una semana,este cambio de horario puede ser causado por dias festivos o bien solicitudes por parte de los capitanes que  participan en el torneo.
	}
		\UCitem{Versión}{1.0}
		\UCitem{Actor}{Coordinador}
		\UCitem{Propósito}{Permitir actualizar la fecha de un partido segun las necesidades de los participantes del torneo}
		\UCitem{Entradas}{Partido, Nueva fecha, Nuevo horario}
		\UCitem{Origen}{Solicitar cambio de horario de partido}
		\UCitem{Salidas}{Mensaje de confirmación}
		\UCitem{Destino}{Se actualizaran las fechas de los partidos correspondientes}
		\UCitem{Precondiciones}{Tener al menos una solicitud de cambio de fecha}
		\UCitem{Postcondiciones}{Se actualizaran las fechas correspondientes}
		\UCitem{Errores}{}
		\UCitem{Tipo}{Caso de uso primario}
		\UCitem{Observaciones}{Debido a que los torneos se realizan para la comunidad de ESCOM, es normal que soliciten un cambio de horario, debido a los estudios o fechas escolares}
		\UCitem{Autor}{Miguel Ángel Morales García.}
		\UCitem{Revisor}{Nadie}
	\end{UseCase}
	\newpage
	\begin{UCtrayectoria}{Principal}
	\UCpaso[\UCactor] Presiona la pestaña de navegación \IUbutton{Actualizar fecha de partido} en la \IUref{IU51}{Menu del coordinador}
	\UCpaso Despliega la pantalla \IUref{IU2}{Actualizar fecha de partido}
	\UCpaso[\UCactor] De la lista de partidos proximos, selecciona el partido al cual le desea cambiar la fecha}		
	\UCpaso[\UCactor] Selecciona la nueva fecha del partido seleccionado}
	\UCpaso[\UCactor] Selecciona la nueva hora de la fecha seleccionada}
	\UCpaso[\UCactor] Presiona el boton \IUbutton{Modificar}
	\UCpaso Fecha modificada.
	\UCpaso Termina el caso de uso.
\end{UCtrayectoria}

\begin{UCtrayectoriaA}{A}{Modificar}
	\UCpaso Muestra el mensaje {\bf MSG4.1-}''Seguro que quieres modificar la fecha?''.
	\UCpaso[\UCactor] Presiona el boton \IUbutton{Aceptar}
	\UCpaso Muestra el mensaje {\bf MSG4.2-}''Fecha modificada''.
	\UCpaso Termina el caso de uso.
	\end{UCtrayectoriaA}

	\begin{UCtrayectoriaA}{B}{Cancelar Modificación}
		\UCpaso[\UCactor] Presiona el botón \IUbutton{Cancelar} en el mensaje {\bf MSG4.1-}''Seguro que quieres modificar la fecha?''}
		\UCpaso Cancela modificación
	\end{UCtrayectoriaA}
