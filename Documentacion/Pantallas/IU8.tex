\subsection{IU8.0 Pantalla de consulta de grupos y estadisticas.}

\subsubsection{Objetivo}
	Los usuarios pueden consultar los grupos y estadisticas de los torneos.

\subsubsection{Diseño}
\IUfig[.6]{gui/IU8}{IU8}{Pantalla de consulta de grupos y estadisticas.}
  El actor puede consultar los grupos y estadisticas de los torneos en curso.


\subsubsection{Opciones}
\begin{itemize}
	\item \IUbutton{Grupo A}: Muestra los datos del grupo A.
	\item \IUbutton{Grupo B}: Muestra los datos del grupo B.
	\item \IUbutton{Grupo C}: Muestra los datos del grupo C.
	\item \IUbutton{Grupo D}: Muestra los datos del grupo D.
	\item \IUbutton{Rapido}: Muestra los datos correspondientes al torneo de futbol Rápido.
	\item \IUbutton{Asocioacion}: Muestra los datos correspondientes al torneo de futbol Asociación.
\end{itemize}
